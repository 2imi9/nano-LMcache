% nano-LMcache architecture — "Attention Is All You Need"-style block diagram.
% Build:  pdflatex figure.tex   (PDF is cropped by the standalone class)
\documentclass[border=14pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{arrows.meta, positioning, fit}
\begin{document}
\begin{tikzpicture}[
  font=\sffamily\small, >=Stealth,
  box/.style={rounded corners=3pt, draw=black!70, thick, align=center,
              minimum height=11mm, minimum width=34mm, inner sep=3pt},
  % color = meaning: green = cache HIT (reused, cheap), red = MISS (recompute, costly),
  % neutral gray = ordinary pipeline step, blue = the KV memory.
  neutral/.style={box, fill=black!5},
  hashb/.style={neutral},
  lookb/.style={neutral},
  mrgb/.style={neutral},
  loadb/.style={box, fill=green!16},
  compb/.style={box, fill=red!13},
  storeb/.style={box, fill=blue!8, minimum height=14mm},
  io/.style={align=center, font=\sffamily\small\bfseries},
  arr/.style={-{Stealth[length=2.6mm]}, thick, black!78},
  darr/.style={{Stealth[length=2.6mm]}-{Stealth[length=2.6mm]}, thick, black!60},
]

% ---- main vertical pipeline (explicit coords, generous spacing) ----
\node[io]    (tok)  at (0,0)   {Prompt token ids};
\node[hashb] (hash) at (0,1.8) {Split into chunks\\ $+$ chained prefix hash};
\node[lookb] (look) at (0,3.7) {Prefix lookup\\ (cache manager)};
\node[loadb] (load) at (-2.6,6.0) {Reuse cached KV\\ \textit{hit:} skip prefill};
\node[compb] (comp) at ( 2.6,6.0) {Compute prefill\\ \textit{miss:} attention};
\node[mrgb]  (asm)  at (0,8.9) {Assemble KV\\ (reused $+$ new)};
\node[io]    (dec)  at (0,10.7) {Decode \& serve};

% ---- cache store: connects ONLY to the lookup node (no crossing arrows) ----
\node[storeb] (kv) at (5.6,3.7) {CPU KV store\\ hash $\to$ KV chunk\\ (LRU, bounded)};

% ---- arrows ----
\draw[arr] (tok)  -- (hash);
\draw[arr] (hash) -- (look);
\draw[arr, green!50!black] (look) -- (load) node[pos=0.55, above, sloped, font=\scriptsize]{hit};
\draw[arr, red!60!black]   (look) -- (comp) node[pos=0.55, above, sloped, font=\scriptsize]{miss};
\draw[arr, green!50!black] (load) -- (asm);
\draw[arr, red!60!black]   (comp) -- (asm);
\draw[arr] (asm)  -- (dec);
\draw[darr] (look) -- (kv) node[midway, above, font=\scriptsize]{load / store};

% ---- dashed group box: the nano-LMcache layer ----
\node[draw=black!45, dashed, rounded corners=8pt, fill=none,
      fit=(hash)(look)(load)(comp)(kv), inner sep=6mm,
      label={[black!55, font=\sffamily\footnotesize\bfseries]above:nano-LMcache \textmd{(prefix-cache layer)}}] {};

\end{tikzpicture}
\end{document}
